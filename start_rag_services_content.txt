<#
.SYNOPSIS
    Unified startup script for Antigravity RAG Services.
    Starts ChromaDB Server (8000) and Embedding Server (8001).

.DESCRIPTION
    This script ensures the RAG memory system is running.
    It handles:
    1. Checking if ports 8000/8001 are free.
    2. Starting the ChromaDB server (Database).
    3. Starting the Embedding server (AI Model).
    4. Waiting for health checks to pass.
    
    Portability:
    - Data is stored in $HOME\.gemini\antigravity\.rag\chroma_db
    - To move to another machine, just copy the .gemini folder and run this script.

.EXAMPLE
    .\start_rag_services.ps1
#>

$ErrorActionPreference = "Stop"
# Fix encoding for Cyrillic
[Console]::OutputEncoding = [System.Text.Encoding]::UTF8

# Configuration
$RAG_ROOT = "$PSScriptRoot"
$CHROMA_DB_PATH = "$env:USERPROFILE\.gemini\antigravity\.rag\chroma_db"
$CHROMA_PORT = 8000
$EMBED_PORT = 8001

Write-Host "üöÄ –ó–∞–ø—É—Å–∫ —Å–µ—Ä–≤—ñ—Å—ñ–≤ Antigravity RAG..." -ForegroundColor Cyan

# 1. Start ChromaDB Server
Write-Host "1Ô∏è‚É£  –ó–∞–ø—É—Å–∫ –±–∞–∑–∏ –¥–∞–Ω–∏—Ö (ChromaDB –Ω–∞ –ø–æ—Ä—Ç—É $CHROMA_PORT)..." -ForegroundColor Green
if (Get-NetTCPConnection -LocalPort $CHROMA_PORT -ErrorAction SilentlyContinue) {
    Write-Host "   ‚ö†Ô∏è  –ü–æ—Ä—Ç $CHROMA_PORT –≤–∂–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î—Ç—å—Å—è. –í–≤–∞–∂–∞—î–º–æ, —â–æ ChromaDB –≤–∂–µ –ø—Ä–∞—Ü—é—î." -ForegroundColor Yellow
}
else {
    $chromaArgs = "run", "--path", "$CHROMA_DB_PATH", "--port", "$CHROMA_PORT", "--host", "localhost"
    $chromaProcess = Start-Process -FilePath "chroma" -ArgumentList $chromaArgs -PassThru -WindowStyle Hidden
    Write-Host "   ‚úÖ ChromaDB –∑–∞–ø—É—â–µ–Ω–æ (PID: $($chromaProcess.Id))"
}

# 2. Start Embedding Server
Write-Host "2Ô∏è‚É£  –ó–∞–ø—É—Å–∫ AI –º–æ–¥–µ–ª—ñ (Embedding Server –Ω–∞ –ø–æ—Ä—Ç—É $EMBED_PORT)..." -ForegroundColor Green
if (Get-NetTCPConnection -LocalPort $EMBED_PORT -ErrorAction SilentlyContinue) {
    Write-Host "   ‚ö†Ô∏è  –ü–æ—Ä—Ç $EMBED_PORT –≤–∂–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î—Ç—å—Å—è. –í–≤–∞–∂–∞—î–º–æ, —â–æ Embedding Server –≤–∂–µ –ø—Ä–∞—Ü—é—î." -ForegroundColor Yellow
}
else {
    $embedScript = Join-Path $RAG_ROOT "embedding_server.py"
    $embedProcess = Start-Process -FilePath "python" -ArgumentList "$embedScript" -PassThru -WindowStyle Hidden
    Write-Host "   ‚úÖ Embedding Server –∑–∞–ø—É—â–µ–Ω–æ (PID: $($embedProcess.Id))"
    Write-Host "   ‚è≥ –û—á—ñ–∫—É–≤–∞–Ω–Ω—è –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—è –º–æ–¥–µ–ª—ñ (~30-60—Å)..." -ForegroundColor Gray
}

# 3. Health Checks
Write-Host "3Ô∏è‚É£  –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ —Å—Ç–∞—Ç—É—Å—É..." -ForegroundColor Green

# Check Embedding Server
$retries = 30
$embedReady = $false
Do {
    Try {
        $resp = Invoke-RestMethod -Uri "http://localhost:$EMBED_PORT/health" -Method Get -ErrorAction Stop
        if ($resp.status -eq "ok") { $embedReady = $true }
    }
    Catch {
        Write-Host "." -NoNewline
        Start-Sleep -Seconds 2
        $retries--
    }
} While (-not $embedReady -and $retries -gt 0)

Write-Host ""
if ($embedReady) {
    Write-Host "   ‚úÖ AI –ú–æ–¥–µ–ª—å –ì–û–¢–û–í–ê!" -ForegroundColor Green
}
else {
    Write-Host "   ‚ùå AI –ú–æ–¥–µ–ª—å –Ω–µ –≤—ñ–¥–ø–æ–≤—ñ–¥–∞—î. –ü–µ—Ä–µ–≤—ñ—Ä—Ç–µ –ª–æ–≥–∏." -ForegroundColor Red
}

# Check ChromaDB
Try {
    $resp = Invoke-RestMethod -Uri "http://localhost:$CHROMA_PORT/api/v1/heartbeat" -Method Get -ErrorAction SilentlyContinue
    Write-Host "   ‚úÖ –ë–∞–∑–∞ –¥–∞–Ω–∏—Ö –ì–û–¢–û–í–ê!" -ForegroundColor Green
}
Catch {
    Write-Host "   ‚ö†Ô∏è  –ë–∞–∑–∞ –¥–∞–Ω–∏—Ö, –º–æ–∂–ª–∏–≤–æ, —â–µ —ñ–Ω—ñ—Ü—ñ–∞–ª—ñ–∑—É—î—Ç—å—Å—è..." -ForegroundColor Yellow
}

Write-Host ""
Write-Host "üéâ –°–∏—Å—Ç–µ–º–∞ RAG –ø–æ–≤–Ω—ñ—Å—Ç—é –≥–æ—Ç–æ–≤–∞ –¥–æ —Ä–æ–±–æ—Ç–∏!" -ForegroundColor Cyan
Write-Host "–¢–µ–ø–µ—Ä –≤–∏ –º–æ–∂–µ—Ç–µ –∑–∞–ø—É—Å–∫–∞—Ç–∏ —Å–≤–æ—ó—Ö –∞–≥–µ–Ω—Ç—ñ–≤."
